<script>

document.querySelectorAll('.tab-button').forEach(btn => {

btn.addEventListener('click', function() {

document.querySelectorAll('.tab-button').forEach(b => b.classList.remove('active'));

this.classList.add('active');


const childId = this.getAttribute('data-child');

document.querySelectorAll('.child-section').forEach(sec => sec.classList.remove('active'));

document.getElementById(childId).classList.add('active');

});

});


document.querySelectorAll('.task-form').forEach(form => {

form.addEventListener('submit', function(e) {

e.preventDefault();

const input = this.querySelector('input[type="text"]');

const taskText = input.value.trim();

if (taskText) {

const li = document.createElement('li');

li.innerHTML = `<input type="checkbox"> ${taskText}`;

this.previousElementSibling.appendChild(li);

input.value = '';

}

});

});

</script>
